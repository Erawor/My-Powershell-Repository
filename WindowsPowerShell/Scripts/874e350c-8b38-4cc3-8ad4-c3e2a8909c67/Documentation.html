function global:Get-WindowsUpdateLog {
#Requires -Version 2.0            
[CmdletBinding()]            
 Param             
   (                       
    [Parameter(Mandatory=$true,
               Position=0,                          
               ValueFromPipeline=$true,            
               ValueFromPipelineByPropertyName=$true)]            
    [String[]]$ComputerName,
    [String]$Filter
   )#End Param

Begin            
{            
 Write-Verbose "`n Checking Windows Update Log . . . "
 $i = 0            
}#Begin          
Process            
{
    $ComputerName | ForEach-Object {
        $Computer = $_
        # Test-OSArchitecture is here: 
        # http://gallery.technet.microsoft.com/scriptcenter/26539b66-13a7-44f6-9adb-886c54fc141f
        $SysDrive = (Test-OSArchitecture -ComputerName $computer).SysDrive
        Get-Content -Path "\\$Computer\$SysDrive\WINDOWS\Windowsupdate.log" | Where-Object {$_ -match $Filter}
    }#foreach(Computername)

}#Process
End
{

}#End

}#Get-WindowsUpdateLog
